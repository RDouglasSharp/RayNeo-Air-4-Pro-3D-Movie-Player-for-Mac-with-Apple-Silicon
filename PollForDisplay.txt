1. Polling for the Target Display
Create a timer that checks connected screens until your target dimensions are available.

swift

import Cocoa
import Combine

class WindowMonitor {
    var cancellable: AnyCancellable?
    
    func pollForDisplay(width: CGFloat, height: CGFloat, completion: @escaping (NSScreen) -> Void) {
        // Poll immediately and then listen for any screen configuration changes
        cancellable = NotificationCenter.default
            .publisher(for: NSApplication.didChangeScreenParametersNotification)
            .merge(with: Just(Notification(name: NSApplication.didChangeScreenParametersNotification)))
            .sink { [weak self] _ in
                if let targetScreen = self?.findScreenWith(width: width, height: height) {
                    completion(targetScreen)
                    self?.cancellable?.cancel() // Stop polling once found
                }
            }
    }
    
    private func findScreenWith(width: CGFloat, height: CGFloat) -> NSScreen? {
        return NSScreen.screens.first { screen in
            // Compare dimensions, accounting for possible fractional points
            return abs(screen.visibleFrame.width - width) < 1.0 && 
                   abs(screen.visibleFrame.height - height) < 1.0
        }
    }
}

2. Moving Your Output to the Screen
Once your target screen is found, you can move your NSWindow to that display’s coordinates.

swift

func moveWindow(_ window: NSWindow, toScreen screen: NSScreen) {
    var frame = window.frame
    // Set the origin of the window to the bottom-left of the target screen
    frame.origin = screen.visibleFrame.origin
    
    window.setFrame(frame, display: true, animate: true)
}
